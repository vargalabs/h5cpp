70 template <class T, class... args_t>
71 inline h5::ds_t write( const h5::ds_t& ds, const T* ptr, args_t&&... args ) try {
72 // element types: pod | [signed|unsigned](int8 | int16 | int32 | int64) | float | double
73 using tcount = typename arg::tpos<const h5::count_t&,const args_t&...>;
74 using toffset = typename arg::tpos<const h5::offset_t&, const args_t&...>;
75 using tstride = typename arg::tpos<const h5::stride_t&, const args_t&...>;
76 using tblock = typename arg::tpos<const h5::block_t&, const args_t&...>;
77 static_assert( tcount::present,"h5::count_t{ ... } must be provided to describe T* memory region" );
78 //static_assert( utils::is_supported<T>, "error: " H5CPP_supported_elementary_types );
79
80 auto tuple = std::forward_as_tuple(args...);
81 h5::count_t count = std::get<tcount::value>( tuple ); //we make a copy of it
82 const h5::dxpl_t& dxpl = arg::get(h5::default_dxpl, args...); // gets reference to default value if property not present
83 h5::sp_t file_space{H5Dget_space( static_cast<::hid_t>(ds) )};
84
85 int rank = h5::get_simple_extent_ndims( file_space );
86 hsize_t n_elements = static_cast<hsize_t>(count);
87 hid_t dapl = h5::get_access_plist( ds );
88 herr_t err = 0;
89
90 // The high_throughput pipeline uses H5Dwrite_chunk, which requires a
91 // chunked dataset layout. Guard on H5D_CHUNKED so a DAPL with the flag
92 // applied to a contiguous/compact dataset falls through to standard
93 // H5Dwrite rather than triggering "not a chunked dataset" errors
94 // (which manifest as a SegFault on Windows MSVC, see #242 follow-up).
95 const bool use_pipeline = [&]() {
96 if (!H5Pexist(dapl, H5CPP_DAPL_HIGH_THROUGHPUT)) return false;
97 hid_t dcpl_id = H5Dget_create_plist(static_cast<hid_t>(ds));
98 if (dcpl_id < 0) return false;
99 H5D_layout_t layout = H5Pget_layout(dcpl_id);
100 H5Pclose(dcpl_id);
101 return layout == H5D_CHUNKED;
102 }();
103 if( use_pipeline ){
104 const h5::block_t& block = arg::get( h5::default_block, args...);
105 const h5::offset_t& offset = arg::get( h5::default_offset, args...);
106 const h5::stride_t& stride = arg::get( h5::default_stride, args...);
107
108 h5::impl::pipeline_t<impl::basic_pipeline_t>* filters;
109 H5Pget(dapl, H5CPP_DAPL_HIGH_THROUGHPUT, &filters);
110 filters->write(ds, offset, stride, block, count, dxpl, ptr);
111 } else {
112 h5::sp_t mem_space = h5::create_simple( n_elements );
113 h5::select_all( mem_space );
114 // we want to optimize for best hyper block selection, in order to do that
115 // let's find out what level of control is needed
116 if constexpr (toffset::present || tstride::present || tblock::present){
117 // HYPERBLOCK selection: we either have the argument in `args...` or using default values
118 const h5::block_t& block = arg::get( h5::default_block, args...);
119 const h5::offset_t& offset = arg::get( h5::default_offset, args...);
120 const h5::stride_t& stride = arg::get( h5::default_stride, args...);
121 if constexpr( tblock::present ){ // we have to normalise `count` such that `size[i] = count[i] * block[i]` holds
122 for(int i=0; i < rank; i++) count[i] /= block[i];
123 err = H5Sselect_hyperslab(file_space, H5S_SELECT_SET, *offset, *stride, *count, *block);
124 } else { // we have to convert h5::count_t{..} to h5::block{..} and initiate a single block transfer
125 h5::block_t block_ = static_cast<h5::block_t>(count);
126 block_.rank = rank; // memory space may have different rank, be sure to use the rank of file_space
127 err = H5Sselect_hyperslab(file_space, H5S_SELECT_SET, *offset, *stride, *h5::default_count, *block_);
128 }
129 } else // SELECT_ALL this is the fastest approach, mem_space and file_space must match
130 err = H5Sselect_all(file_space);
131 // throw an exception if eny error
132 H5CPP_CHECK_NZ(err, h5::error::io::dataset::write, h5::error::msg::select_hyperslab);
133 // MSVC partial-ordering bug: the unqualified `::h5::write(ds, mem, file, dxpl, ptr)`
134 // is ambiguous between the inner overload at line 21 and the variadic forwarder at
135 // line 181 (with T=sp_t). Inline the H5Dwrite call here — matches the H5Dread.hpp:73
136 // idiom — so there is no recursive call back into this overload set.
137 using element_t = typename impl::decay<T>::type;
138 h5::meta::resolved_type_t<element_t> type;
139 H5CPP_CHECK_NZ(
140 H5Dwrite( static_cast<hid_t>(ds), static_cast<hid_t>(type),
141 static_cast<hid_t>(mem_space), static_cast<hid_t>(file_space),
142 static_cast<hid_t>(dxpl), ptr),
143 h5::error::io::dataset::write, h5::error::msg::write_dataset);
144 }
145 return ds;
146 } catch ( const std::exception& err ){
147 throw h5::error::io::dataset::write( err.what() );
148 }

204 template <class T, class... args_t,
205 class = std::enable_if_t<!std::is_pointer_v<std::decay_t<T>>>>
206 inline h5::ds_t write(const h5::ds_t& ds, const T& ref, args_t&&... args) try {
207 using tcount = typename arg::tpos<const h5::count_t&,const args_t&...>;
208 using element_t = typename impl::decay<T>::type;
209
210 using traits = h5::meta::access_traits_t<T>;
211 constexpr auto kind = traits::kind;
212
213 if constexpr (kind == h5::meta::access_t::contiguous
214 || kind == h5::meta::access_t::object
215 || kind == h5::meta::access_t::text) {
216 auto ptr = traits::data(ref);
217 if constexpr (tcount::present)
218 ::h5::write(ds, ptr, args...);
219 else {
220 h5::count_t count = traits::size( ref );
221 ::h5::write(ds, ptr, count, args...);
222 }
223 } else if constexpr (kind == h5::meta::access_t::pointers) {
224 // e.g. vector<string>, vector<vector<int>> — elements have .data() but are not flat
225 using element_t = typename impl::decay<typename traits::element_t>::type;
226 std::vector<element_t> elements;
227 const element_t* ptrs = h5::gather(ref, elements);
228 if constexpr (tcount::present)
229 ::h5::write<element_t>(ds, ptrs, args...);
230 else {
231 h5::count_t count = traits::size( ref );
232 ::h5::write<element_t>(ds, ptrs, count, args...);
233 }
234 } else if constexpr (kind == h5::meta::access_t::iterators) {
235 // e.g. list<int>, set<int>, map<K,V> — no direct pointer, copy to staging buffer
236 using element_t = typename impl::decay<typename traits::element_t>::type;
237 auto count = traits::size(ref);
238 size_t n = 1;
239 for (std::size_t i = 0; i < count.size(); ++i) n *= count[i];
240 std::vector<element_t> buffer;
241 buffer.reserve(n);
242 for (const auto& elem : ref)
243 buffer.push_back(elem);
244 ::h5::write(ds, buffer.data(), h5::count_t(count), args...);
245 } else {
246 static_assert(kind != h5::meta::access_t::unsupported,
247 "unsupported type for h5::write");
248 }
249 return ds;
250 } catch ( const std::exception& err ){
251 throw h5::error::io::dataset::write( err.what() );
252 }

300 template <class T, class... args_t>
301 inline h5::ds_t write( const h5::fd_t& fd, const std::string& dataset_path, const T* ptr, args_t&&... args ){
302 using tcount = typename arg::tpos<const h5::count_t&, const args_t&...>;
303 static_assert( tcount::present, "h5::count_t{ ... } must be provided to describe T* memory region" );
304 h5::ds_t ds; // initialized to H5I_UNINIT
305
306 h5::mute(); // find out if we have to create the dataset
307 bool is_dataset_present = H5Lexists(fd, dataset_path.c_str(), H5P_DEFAULT) > 0;
308 h5::unmute(); // <- make sure not to mute error handling longer than needed
309 if (is_dataset_present) {
310 const h5::dapl_t& dapl = arg::get(h5::default_dapl, args...);
311 ds = h5::open(fd, dataset_path, dapl);
312 } else {
313 // dataset doesn't exist, or some error happened, since h5::create doesn't know of the
314 // memory space size as `T& ref` never passed along we have to compute the `h5::current_dims_t{}` upfront
315 using tcurrent_dims = typename arg::tpos<const h5::current_dims_t&, const args_t&...>;
316 if constexpr (tcurrent_dims::present) // user knows what he is doing, specified h5::current_dims{} explicitly
317 ds = h5::create<T>(fd, dataset_path, args...);
318 else {
319 h5::current_dims_t current_dims = static_cast<h5::current_dims_t>(std::get<tcount::value>(std::forward_as_tuple(args...))); // count must be present, we checked
320 ds = h5::create<T>(fd, dataset_path, current_dims, args...); // and use it to create dataset
321 }
322 }
323 // we either have `ds` != H5I_UNINIT or an exception thrown, safe to delegate
324 return ::h5::write(ds, ptr, args...);
325 }

376 h5::ds_t ds; // initialized to H5I_UNINIT
377 // find out if we have to create the dataset
378 h5::mute();
379 // Returns a negative value when the function fails and may return a negative value if the link does not exist.
380 // - name is not local to the group specified by loc_id or, if loc_id is something other than a group identifier,
381 // name is not local to the root group
382 // - Any element of the relative path or absolute path in name, except the target link, does not exist.
383 bool is_dataset_present = H5Lexists(fd, dataset_path.c_str(), H5P_DEFAULT) > 0;
384 h5::unmute(); // <- make sure not to mute error handling longer than needed
385
386 if (is_dataset_present) {
387 const h5::dapl_t& dapl = arg::get(h5::default_dapl, args...);
388 ds = h5::open(fd, dataset_path, dapl);
389 } else {
390 // dataset doesn't exist, or some error happened, since h5::create doesn't know of the
391 // memory space size as `T& ref` never passed along we have to compute the `h5::current_dims_t{}` upfront
392 using tcurrent_dims = typename arg::tpos<const h5::current_dims_t&, const args_t&...>;
393 using element_t = typename h5::impl::decay<T>::type;
394 if constexpr (tcurrent_dims::present) // user knows what he is doing, specified h5::current_dims{} explicitly
395 ds = h5::create<element_t>(fd, dataset_path, args...);
396 else { // h5::current_dims{..} is explicitly given by `h5::count` and optional h5::offset{}, h5::stride{}, h5::block{}
397 h5::count_t count = impl::size(ref);
398 h5::current_dims_t current_dims = h5::impl::get_current_dims(count, args...); // get correct dimensions
399 ds = h5::create<element_t>(fd, dataset_path, current_dims, args...); // and use it to create dataset
400 }
401 }
402 // we either have `ds` != H5I_UNINIT or an exception thrown, safe to delegate
