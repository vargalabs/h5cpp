47 template<class T, class... args_t> inline
48 h5::ds_t create( const h5::fd_t& fd, const std::string& dataset_path, args_t&&... args ) try {
49 // compile time check of property lists:
50 using tcurrent_dims = typename arg::tpos<const h5::current_dims_t&,const args_t&...>;
51 using tmax_dims = typename arg::tpos<const h5::max_dims_t&,const args_t&...>;
52 using tdcpl = typename arg::tpos<const h5::dcpl_t&,const args_t&...>;
53 using tdt_t = typename arg::tpos<const h5::dt_t<T>&,const args_t&...>;
54
55 //TODO: make copy of default dcpl
56 h5::dcpl_t default_dcpl{ H5Pcreate(H5P_DATASET_CREATE) };
57 // get references to property lists or system default values
58 const h5::lcpl_t& lcpl = arg::get(h5::default_lcpl, args...);
59 const h5::dcpl_t& dcpl = arg::get(default_dcpl, args...);
60 const h5::dapl_t& dapl = arg::get(h5::default_dapl, args...);
61
62 H5CPP_CHECK_PROP( lcpl, h5::error::property_list::misc, "invalid list control property" );
63 H5CPP_CHECK_PROP( dcpl, h5::error::property_list::misc, "invalid data control property" );
64 H5CPP_CHECK_PROP( dapl, h5::error::property_list::misc, "invalid data access property" );
65 // and dimensions
66 h5::current_dims_t current_dims_default{0}; // if no current dims_present
67 // this mutable value will be referenced
68 const h5::current_dims_t& current_dims = arg::get(current_dims_default, args...);
69 h5::max_dims_t max_dims_default{0};
70 const h5::max_dims_t& max_dims = arg::get(max_dims_default, args... );
71 bool has_unlimited_dimension = false;
72 size_t rank = 0;
73 h5::sp_t space_id{H5I_UNINIT}; // set to invalid state
74 h5::ds_t ds{H5I_UNINIT};
75
76 if constexpr( tmax_dims::present ){
77 rank = max_dims.size();
78 if constexpr( !tcurrent_dims::present ){
79 // set current dimensions to given one or zero if H5S_UNLIMITED mimics matlab(tm) behavior while allowing extendable matrices
80 for(hsize_t i=0; i<rank; i++)
81 current_dims_default[i] = max_dims[i] != H5S_UNLIMITED
82 ? max_dims[i] : (has_unlimited_dimension=true, static_cast<hsize_t>(0));
83 current_dims_default.rank = static_cast<int>(rank);
84 }
85 } else static_assert( tcurrent_dims::present,"current or max dimensions must be present in order to create a dataset!" );
86
87 if( !tdcpl::present && ( has_unlimited_dimension || (tcurrent_dims::present && tmax_dims::present)) ){
88 chunk_t chunk{0}; chunk.rank = current_dims.rank;
89 for(hsize_t i=0; i<rank; i++)
90 chunk[i] = current_dims[i] ? current_dims[i] : 1;
91 h5::set_chunk(default_dcpl, chunk );
92 }
93 // use move semantics to set space
94 space_id = tmax_dims::present ?
95 std::move( h5::create_simple( current_dims, max_dims ) ) : std::move( h5::create_simple( current_dims ) );
96 using element_t = typename meta::decay<T>::type;
97 h5::meta::resolved_type_t<element_t> type;
98 if constexpr (tdt_t::present)
99 type = static_cast<::hid_t>(
100 std::get<tdt_t::value>( std::forward_as_tuple( args...)));
101 return h5::createds(fd, dataset_path, static_cast<::hid_t>(type), space_id, lcpl, dcpl, dapl);
102 } catch( const std::runtime_error& err ) {
103 throw h5::error::io::dataset::create( err.what() );
104 }
