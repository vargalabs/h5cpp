17 /* proxy object that gets converted to property_id with restriction that
18 * only same class properties may be daisy chained */
19 template <class Derived, class phid_t>
20 struct prop_base {
21 using hidtype = phid_t;
22 using type = phid_t;
23
24 prop_base(){ }
25 // removed ctor
26 ~prop_base(){
27 if( H5Iis_valid( handle ) ){
28 H5Pclose( handle );
29 }
30 }
31 // prevents property type mismatch at compile time:
32 template <class R>
33 std::enable_if_t< std::is_same_v<typename R::hidtype, hidtype> ,
34 const prop_base<Derived, phid_t>& >
35 operator|( const R& rhs ) const {
36 rhs.copy( handle );
37 return *this;
38 }
39 // chain with an existing property-list handle (e.g. h5::chunk{N} | dcpl):
40 // deep-copy rhs via H5Pcopy, apply this property onto the copy, return owning handle.
41 template <class R>
42 std::enable_if_t< std::is_same_v<R, phid_t>, phid_t >
43 operator|( const R& rhs ) const {
44 ::hid_t merged;
45 H5CPP_CHECK_NZ( (merged = H5Pcopy( static_cast<::hid_t>(rhs) )),
46 h5::error::property_list::misc, "failed to copy property list");
47 static_cast<const Derived*>(this)->copy_impl( merged );
48 return phid_t{ merged };
49 }
50 // convert to propery
51 void copy(::hid_t handle_) const { /*CRTP idiom*/
52 static_cast<const Derived*>(this)->copy_impl( handle_ );
53 }
54 /*transfering ownership to managed handle*/
55 operator phid_t( ) const {
56 static_cast<const Derived*>(this)->copy_impl( handle );
57 H5Iinc_ref( handle ); /*keep this alive */
58 return phid_t{handle};
59 }
60 ::hid_t handle;
61 };

64 /*property with a capi function call with some arguments, also is the base for other properties */
65 template <class phid_t, defprop_t init, class capi, typename capi::fn_t capi_call>
66 struct prop_t : prop_base< prop_t<phid_t,init,capi,capi_call>, phid_t > {
67 using base_t = prop_base<prop_t<phid_t,init,capi,capi_call>, phid_t>;
68 using args_t = typename capi::args_t;
69 using capi_t = typename capi::type;
70 using type = phid_t;
71
72 prop_t( typename capi::args_t values ) : args( values ) {
73 H5CPP_CHECK_NZ( (this->handle = H5Pcreate(init())),
74 h5::error::property_list::misc, "failed to create property");
75 }
76 prop_t(){
77 H5CPP_CHECK_NZ( (this->handle = H5Pcreate(init())),
78 h5::error::property_list::misc, "failed to create property");
79 }
80 void copy_impl(::hid_t id) const {
81 //int i = capi_call + 1;
82 /*CAPI needs `this` hid_t id passed along */
83 capi_t capi_args = std::tuple_cat( std::tie(id), args );
84 H5CPP_CHECK_NZ( std::apply(capi_call, capi_args),
85 h5::error::property_list::argument,"failed to parse arguments...");
86 }
87
88 args_t args;
89 };

91 /* T prop is much similar to other properties, except the value needs HDF5 type info and passed by
92 * const void* pointer type. This is reflected by templating `prop_t`
93 */
94 template <class phid_t, defprop_t init, class capi, typename capi::fn_t capi_call, class T>
95 struct tprop_t final : prop_t<phid_t,init,capi,capi_call> { // ::hid_t,const void*
96 using type = phid_t;
97 using base_t = prop_t<phid_t,init,capi,capi_call>;
98
99 tprop_t( T value ) : base_t( std::make_tuple( h5::copy<T>( h5::dt_t<T>() ), &this->value) ), value(value) {
100 }
101 ~tprop_t(){ // don't forget that the first argument is a HDF5 type info, that needs closing
102 if( H5Iis_valid( std::get<0>(this->args) ) )
103 H5Tclose( std::get<0>(this->args) );
104 }
105
106 T value;
107 };
108 /* takes an arbitrary length of hsize_t sequence and calls H5Pset_xxx
109 */
110 template <class phid_t, defprop_t init, class capi, typename capi::fn_t capi_call>
111 struct aprop_t final : prop_t<phid_t,init,capi,capi_call> {
112 using type = phid_t;
113 using base_t = prop_t<phid_t,init,capi,capi_call>;
114
115 aprop_t( std::initializer_list<hsize_t> values )
116 : base_t( std::make_tuple( static_cast<int>(values.size()), this->values) ) {
117 std::copy( std::begin(values), std::end(values), this->values);
118 }
119 template <class T, size_t N>
120 aprop_t( std::array<T,N> values )
121 : base_t( std::make_tuple( static_cast<int>(values.size()), this->values) ) {
122 std::copy( std::begin(values), std::end(values), this->values);
123 }
124 hsize_t values[H5CPP_MAX_RANK];
125 };
