2// Doorbell: a sequenced wake primitive. C++20 uses atomic wait/notify
3// (lock-free, maps to OS futex). C++17 falls back to mutex+condvar.
4// API is the same either way.
5#include <cstdint>
6#ifdef __cpp_lib_atomic_wait

8namespace h5::detail {
9 struct doorbell_t {
10 void ring() noexcept {
11 seq_.fetch_add(1, std::memory_order_release);
12 seq_.notify_one();
13 }
14 void ring_all() noexcept {
15 seq_.fetch_add(1, std::memory_order_release);
16 seq_.notify_all();
17 }
18 void wait(std::uint32_t last) const noexcept {
19 seq_.wait(last, std::memory_order_acquire);
20 }
21 std::uint32_t load() const noexcept {
22 return seq_.load(std::memory_order_acquire);
23 }
24 private:
25 std::atomic<std::uint32_t> seq_{0};
26 };
27} // namespace h5::detail

31namespace h5::detail {
32 struct doorbell_t {
33 void ring() noexcept {
34 { std::lock_guard<std::mutex> lk(m_); ++seq_; }
35 cv_.notify_one();
36 }
37 void ring_all() noexcept {
38 { std::lock_guard<std::mutex> lk(m_); ++seq_; }
39 cv_.notify_all();
40 }
41 void wait(std::uint32_t last) {
42 std::unique_lock<std::mutex> lk(m_);
43 cv_.wait(lk, [&]{ return seq_ != last; });
44 }
45 std::uint32_t load() const noexcept {
46 std::lock_guard<std::mutex> lk(const_cast<std::mutex&>(m_));
47 return seq_;
48 }
49 private:
50 mutable std::mutex m_;
51 std::condition_variable cv_;
52 std::uint32_t seq_{0};
53 };
54} // namespace h5::detail
