164 inline h5::ds_t createds(const h5::fd_t& fd, const std::string& path, ::hid_t type_id,
165 const h5::sp_t& sp, const h5::lcpl_t& lcpl, const h5::dcpl_t& dcpl, const h5::dapl_t& dapl ){
166 hid_t ds;
167 H5CPP_CHECK_NZ(( ds = H5Dcreate2( static_cast<hid_t>(fd), path.data(), type_id, static_cast<hid_t>(sp),
168 static_cast<hid_t>(lcpl), static_cast<hid_t>(dcpl), static_cast<hid_t>( dapl ) )),
169 std::runtime_error, h5::error::msg::create_dataset );
170 //FIXME: hack to carry prop over
171 h5::ds_t ds_{ds};
172
173 switch( H5Pget_layout(static_cast<::hid_t>( dcpl)) ){
174 case H5D_LAYOUT_ERROR: break;
175 case H5D_NLAYOUTS: break;
176 case H5D_COMPACT: break;
177 case H5D_CONTIGUOUS: break;
178 case H5D_CHUNKED:
179 if( H5Pexist(dapl, H5CPP_DAPL_HIGH_THROUGHPUT) ){
180 // grab pointer to uninitialized pipeline
181 h5::impl::pipeline_t<impl::basic_pipeline_t>* ptr;
182 H5Pget(dapl, H5CPP_DAPL_HIGH_THROUGHPUT, &ptr);
183 hid_t elem_type_id = H5Dget_type( static_cast<::hid_t>(ds) );
184 size_t element_size = H5Tget_size( elem_type_id );
185 ptr->set_cache(dcpl, element_size);
186 }
187 break;
188 case H5D_VIRTUAL: break;
189 }
190 ds_.dapl = static_cast<::hid_t>( dapl );
191 return ds_;
192 }

200 inline void * get_fill_value(const h5::dcpl_t& dcpl, const h5::dt_t<void*>& type, size_t size){
201
202 H5D_fill_value_t fill_value_status;
