99 // Largest possible output across all supported filters for a given input size.
100 // Use this to size the pipeline scratch buffers.
101 inline size_t filter_scratch_bound(size_t size) {
102 size_t b = deflate_bound(size);
103 b = std::max(b, lz4_bound(size));
104 b = std::max(b, zstd_bound(size));
105 b = std::max(b, size + 4); // fletcher32 appends 4-byte checksum
106 return b;
107 }
108 inline size_t zlib_deflate_encode(void* dst, const void* src, size_t size, unsigned level) {
109#if defined(H5CPP_HAS_LIBDEFLATE)
110 libdeflate_compressor* compressor = libdeflate_alloc_compressor(static_cast<int>(level));
111 if (!compressor)
112 return 0;
113 const size_t nbytes = libdeflate_zlib_compress(
114 compressor, src, size, dst, deflate_bound(size));
115 libdeflate_free_compressor(compressor);
116 return nbytes;
117#else
118 uLongf nbytes = static_cast<uLongf>(deflate_bound(size));
119 const int status = compress2(
120 static_cast<Bytef*>(dst), &nbytes,
121 static_cast<const Bytef*>(src), static_cast<uLong>(size),
122 static_cast<int>(level));
123 return status == Z_OK ? static_cast<size_t>(nbytes) : 0;
124#endif
125 }
126 inline size_t zlib_deflate_decode(void* dst, const void* src, size_t compressed_size, size_t output_size) {
127#if defined(H5CPP_HAS_LIBDEFLATE)
128 libdeflate_decompressor* decompressor = libdeflate_alloc_decompressor();
129 if (!decompressor)
130 return 0;
131 size_t actual_size = 0;
132 const libdeflate_result status = libdeflate_zlib_decompress(
133 decompressor, src, compressed_size, dst, output_size, &actual_size);
134 libdeflate_free_decompressor(decompressor);
135 return status == LIBDEFLATE_SUCCESS ? actual_size : 0;
136#else
137 uLongf actual_size = static_cast<uLongf>(output_size);
138 const int status = uncompress(
139 static_cast<Bytef*>(dst), &actual_size,
140 static_cast<const Bytef*>(src), static_cast<uLong>(compressed_size));
141 return status == Z_OK ? static_cast<size_t>(actual_size) : 0;
142#endif
143 }

160 // Byte-level shuffle / unshuffle.
161 // Matching HDF5 H5Z_FILTER_SHUFFLE semantics: for n elements of `type_size` bytes
162 // each, reorder bytes so that byte-k of every element is contiguous.
163 // params[0] = element size in bytes (set by H5Pset_shuffle / H5Pget_filter2).
164 inline size_t shuffle( void* dst, const void* src, size_t size, unsigned flags, size_t n, const unsigned params[] ){
165 const size_t type_size = (n > 0 && params[0] > 1) ? static_cast<size_t>(params[0]) : 1;
166 if (type_size == 1 || size == 0) {
167 memcpy(dst, src, size);
168 return size;
169 }
170 const size_t count = size / type_size;
171 const char* s = static_cast<const char*>(src);
172 char* d = static_cast<char*>(dst);
173 if (flags & H5Z_FLAG_REVERSE) {
174 // Unshuffle: interleaved → original AoS layout
175 for (size_t byte = 0; byte < type_size; ++byte)
176 for (size_t elem = 0; elem < count; ++elem)
177 d[elem * type_size + byte] = s[byte * count + elem];
178 } else {
179 // Shuffle: AoS → byte-plane layout
180 for (size_t byte = 0; byte < type_size; ++byte)
181 for (size_t elem = 0; elem < count; ++elem)
182 d[byte * count + elem] = s[elem * type_size + byte];
183 }
184 return size;
185 }

188 inline uint32_t fletcher32_checksum(const void* data, size_t nbytes) {
189 const uint8_t* p = static_cast<const uint8_t*>(data);
190 uint32_t sum1 = 0, sum2 = 0;
191 // Process 16-bit words; if nbytes is odd the final byte is treated as a
192 // zero-padded 16-bit word (matching HDF5's reference implementation).
193 const size_t words = nbytes / 2;
194 for (size_t i = 0; i < words; ++i) {
195 const uint16_t w = (static_cast<uint16_t>(p[2*i]) << 8) | p[2*i+1];
196 sum1 = (sum1 + w) % 65535u;
197 sum2 = (sum2 + sum1) % 65535u;
198 }
199 if (nbytes & 1u) {
200 const uint16_t w = static_cast<uint16_t>(p[nbytes - 1]) << 8;
201 sum1 = (sum1 + w) % 65535u;
202 sum2 = (sum2 + sum1) % 65535u;
203 }
204 return (sum2 << 16) | sum1;
205 }

208 // Fletcher32 checksum filter.
209 // Encode: appends a 4-byte big-endian checksum → returns size + 4.
210 // Decode: verifies, strips 4-byte checksum → returns size - 4, or 0 on mismatch.
211 inline size_t fletcher32( void* dst, const void* src, size_t size, unsigned flags, size_t n, const unsigned params[]){
212 (void)n; (void)params;
213 if (flags & H5Z_FLAG_REVERSE) {
214 if (size < 4) return 0;
215 const size_t data_size = size - 4;
216 const uint8_t* stored_bytes = static_cast<const uint8_t*>(src) + data_size;
217 const uint32_t stored =
218 (static_cast<uint32_t>(stored_bytes[0]) << 24) |
219 (static_cast<uint32_t>(stored_bytes[1]) << 16) |
220 (static_cast<uint32_t>(stored_bytes[2]) << 8) |
221 static_cast<uint32_t>(stored_bytes[3]);
222 const uint32_t computed = fletcher32_checksum(src, data_size);
223 if (stored != computed) return 0;
224 memcpy(dst, src, data_size);
225 return data_size;
226 } else {
227 const uint32_t checksum = fletcher32_checksum(src, size);
228 memcpy(dst, src, size);
229 uint8_t* out = static_cast<uint8_t*>(dst) + size;
230 out[0] = static_cast<uint8_t>(checksum >> 24);
231 out[1] = static_cast<uint8_t>(checksum >> 16);
232 out[2] = static_cast<uint8_t>(checksum >> 8);
233 out[3] = static_cast<uint8_t>(checksum);
234 return size + 4;
235 }
236 }

240 // H5Z_FILTER_SZIP (id=4): Rice/Golomb lossless compression via vendored libaec/szip.
241 // HDF5 cd_values layout (from H5Zszip.c):
242 // params[0] = options_mask (SZ_EC_OPTION_MASK | SZ_NN_OPTION_MASK | SZ_MSB/LSB | ...)
243 // params[1] = bits_per_pixel (element size in bits, 1..24 or 32/64)
244 // params[2] = pixels_per_block (2..32, must be even)
245 // params[3] = pixels_per_scanline (= total elements in chunk)
246 inline size_t szip( void* dst, const void* src, size_t size, unsigned flags, size_t n, const unsigned params[]){
247 (void)flags; (void)n; (void)params;
248#if defined(H5CPP_HAS_SZIP)
249 if (n < 4) return 0;
250
251 SZ_com_t param;
252 param.options_mask = static_cast<int>(params[0]);
253 param.bits_per_pixel = static_cast<int>(params[1]);
254 param.pixels_per_block = static_cast<int>(params[2]);
255 param.pixels_per_scanline = static_cast<int>(params[3]);
256
257 if (flags & H5Z_FLAG_REVERSE) {
258 // Decompress: output is the original uncompressed chunk.
259 // SZ_BufftoBuffDecompress needs to know the number of output pixels:
260 // pixels = params[3] (pixels_per_scanline stores total chunk elements for HDF5).
261 size_t out_size = static_cast<size_t>(param.pixels_per_scanline)
262 * ((static_cast<size_t>(param.bits_per_pixel) + 7) / 8);
263 int ret = SZ_BufftoBuffDecompress(dst, &out_size, src, size, &param);
264 return (ret == SZ_OK) ? out_size : 0;
265 } else {
266 // Compress.
267 size_t out_size = size + size / 2 + 128; // szip worst-case is ~1.5x
268 if (out_size < size + 128) out_size = size + 128;
269 int ret = SZ_BufftoBuffCompress(dst, &out_size, src, size, &param);
270 return (ret == SZ_OK) ? out_size : 0;
271 }
272#else
273 // Passthrough when szip not compiled in.
274 memcpy(dst, src, size);
275 return size;
276#endif
277 }

306 // Enabled when compiled with H5CPP_HAS_LZ4; falls back to passthrough otherwise.
307 inline size_t lz4( void* dst, const void* src, size_t size, unsigned flags, size_t n, const unsigned params[] ){
308 (void)flags; (void)n; (void)params;
309#if defined(H5CPP_HAS_LZ4)
310 if (flags & H5Z_FLAG_REVERSE) {
311 const int out = LZ4_decompress_safe(
312 static_cast<const char*>(src), static_cast<char*>(dst),
313 static_cast<int>(size),
314 static_cast<int>(decompressed_size_hint(size, n, params)));
315 return out > 0 ? static_cast<size_t>(out) : 0;
316 } else {
317 const int out = LZ4_compress_default(
318 static_cast<const char*>(src), static_cast<char*>(dst),
319 static_cast<int>(size), static_cast<int>(lz4_bound(size)));
320 return out > 0 ? static_cast<size_t>(out) : 0;
321 }
322#else
323 memcpy(dst, src, size);
324 return size;
325#endif
326 }
