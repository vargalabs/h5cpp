31 pt_t& operator=( h5::pt_t&& pt ){
32 // prevent self assign
33 if (this == &pt) return *this;
34 if(H5Iis_valid(this->ds)){ // flush and close dataset
35 this->flush();
36 free(this->fill_value);
37 }
38
39 this->ds = std::move(pt.ds);
40 this->dxpl = std::move(pt.dxpl);
41 this->pipeline = std::move(pt.pipeline);
42
43 this->block_size = pt.block_size;
44 this->element_size = pt.element_size;
45 this->N = pt.N; this->n = pt.n; this->rank = pt.rank;
46 this->ptr = pt.ptr; this->fill_value = pt.fill_value;
47
48 pt.ptr = nullptr; pt.fill_value = nullptr;
49 pt.N=0; pt.n=0; pt.rank=0;
50 for(hsize_t i=0; i<rank; i++){
51 this->offset[i] = pt.offset[i];
52 this->current_dims[i] = pt.current_dims[i];
53 this->chunk_dims[i] = pt.chunk_dims[i];
54 this->count[i] = pt.count[i];
55 }
56 return *this;

112inline
113void h5::pt_t::init( const h5::ds_t& handle ){
114 try {
115 // Re-open with zero HDF5 chunk cache: the default DAPL allocates ~1MB per H5Dopen2
116 // call which accumulates in malloc arenas when pt_t is used in loops, even though
117 // the memory is logically freed on close.
118 hid_t raw = static_cast<hid_t>(handle);
119 hid_t fid = H5Iget_file_id(raw);
120 ssize_t nlen = H5Iget_name(raw, nullptr, 0);
121 std::vector<char> dname(static_cast<size_t>(nlen) + 1);
122 H5Iget_name(raw, dname.data(), dname.size());
123 hid_t dapl = H5Pcreate(H5P_DATASET_ACCESS);
124 H5Pset_chunk_cache(dapl, 0, 0, H5D_CHUNK_CACHE_W0_DEFAULT);
125 ds = h5::ds_t{H5Dopen2(fid, dname.data(), dapl)};
126 H5Pclose(dapl);
127 H5Fclose(fid);
128 dt = h5::dt_t<void>{H5Dget_type(static_cast<hid_t>(ds))};
129 h5::sp_t file_space = h5::get_space( handle );
130 rank = h5::get_simple_extent_dims( file_space, current_dims, nullptr );
131
132 h5::dcpl_t dcpl = h5::get_dcpl( ds );
133 h5::dt_t<void*> type = h5::get_type<void*>( ds );
134 hsize_t size = h5::get_size( type );
135 this->fill_value = h5::get_fill_value(dcpl, type, size);
136 pipeline.set_cache(dcpl, size);
137 this->ptr = pipeline.chunk0;
138 this->block_size = pipeline.block_size;
139 this->element_size = pipeline.element_size;
140 this->N = pipeline.n;
141 pipeline.ds = ds; pipeline.dxpl = dxpl;
142 h5::get_chunk_dims( dcpl, chunk_dims );
143 for(hsize_t i=1; i<rank; i++)
144 current_dims[i] = chunk_dims[i];
145 } catch ( ... ){
146 throw h5::error::io::packet_table::misc( H5CPP_ERROR_MSG("CTOR: unable to create handle from dataset..."));
147 }
148}

159inline void h5::pt_t::append( const std::string& ref ) {
160 static_cast<const char**>( ptr )[n++] = ref.data();
161 if( n != N ) return;
162
163 *offset = *current_dims;
164 *current_dims += *chunk_dims;
165 h5::set_extent(ds, current_dims);
166
167 hsize_t block = 1, count = n;
168 h5::sp_t mem_space{H5Screate_simple(static_cast<int>(rank), &count, nullptr )};
169 h5::sp_t file_space{H5Dget_space( static_cast<::hid_t>(ds) )};
170 h5::select_all( mem_space );
171 H5Sselect_hyperslab( static_cast<hid_t>(file_space), H5S_SELECT_SET, offset, nullptr, &block, &count);
172
173 H5Dwrite( static_cast<hid_t>( ds ),
174 dt, mem_space, file_space, static_cast<hid_t>(dxpl), ptr);
175 n = 0;
176}

213template<class T> inline std::enable_if_t< !h5::meta::is_scalar<T>::value,
214void> h5::pt_t::append( const T& ref ) try {
215 // Iterator-only containers (list, forward_list, deque, set, …) have no .data().
216 // Delegate element-by-element to the scalar append overload.
217 using access = h5::meta::access_traits_t<T>;
218 if constexpr (access::kind == h5::meta::access_t::iterators) {
219 for (const auto& elem : ref)
220 this->append(elem);
221 } else {
222
223 auto dims = meta::size( ref );
224
225 *offset = *current_dims;
226 *current_dims += 1;
227 h5::set_extent(ds, current_dims);
228 auto ptr_ = meta::data( ref );
229 auto dims_ = meta::size( ref );
230
231 switch( dims_.size() ){
232 case 1: // vector
233 if( dims[0] * element_size == block_size )
234 pipeline.write_chunk(offset, block_size, (void*) ptr_ );
235 else throw h5::error::io::packet_table::write(
236 H5CPP_ERROR_MSG("dimension mismatch: "
237 + std::to_string( dims[0] * element_size) + " != " + std::to_string(block_size) ));
238 break;
239 case 2: //matrix
240 if( dims[0] * dims[1] * element_size == block_size )
241 pipeline.write_chunk(offset, block_size, (void*) ptr_ );
242 else throw h5::error::io::packet_table::write(
243 H5CPP_ERROR_MSG("dimension mismatch: "
244 + std::to_string( dims[0] * dims[1] * element_size) + " != " + std::to_string(block_size) ));
245 break;
246 case 3: // cube
247 if( dims[0] * dims[1] * dims[2] * element_size == block_size )
248 pipeline.write_chunk(offset, block_size, (void*) ptr_ );
249 else throw h5::error::io::packet_table::write(
250 H5CPP_ERROR_MSG("dimension mismatch: "
251 + std::to_string( dims[0] * dims[1] * dims[2] * element_size) + " != " + std::to_string(block_size) ));
252 ;break;
253 default:
254 throw h5::error::io::packet_table::misc( H5CPP_ERROR_MSG("objects with rank > 2 are not supported... "));
255 }
256 } // end else (non-iterator path)
257} catch( const std::runtime_error& err ){
258 throw h5::error::io::dataset::append( err.what() );
259}

261inline
262void h5::pt_t::flush(){
263 if( n == 0 ) return;
264 *offset = *current_dims;
265 *current_dims += *chunk_dims;
266 h5::set_extent(ds, current_dims);
267
268 if( H5Tis_variable_str(this->dt)) {
269 hsize_t block = 1, count = n;
270 h5::sp_t mem_space{H5Screate_simple(static_cast<int>(rank), &count, nullptr )};
271 h5::sp_t file_space{H5Dget_space( static_cast<::hid_t>(ds) )};
272 h5::select_all( mem_space );
273 H5Sselect_hyperslab( static_cast<hid_t>(file_space), H5S_SELECT_SET, offset, nullptr, &block, &count);
274
275 H5Dwrite( static_cast<hid_t>( ds ),
276 dt, mem_space, file_space, static_cast<hid_t>(dxpl), ptr);
277 } else {
278 // the remainder of last chunk must be set to fill_value; arbitrary type size supported
279 for(hsize_t i=0; i<(N-n); i++)
280 for(size_t j=0; j < element_size; j++)
281 static_cast<char*>( ptr )[(n + i) * element_size + j] = static_cast<char*>( fill_value )[ j ];
282 pipeline.write_chunk( offset, block_size, ptr );
283 }
284 n = 0;
285}

309inline std::ostream& operator<<(std::ostream &os, const h5::pt_t& pt) {
310 os << std::dec;
311 os <<"packet table:\n"
312 "------------------------------------------\n";
313 if( !H5Iis_valid(pt.ds)) {
314 os << "ds: H5I_UNINIT" <<std::endl;
315 return os;
316 }
317 std::vector<size_t> current_dims(pt.current_dims, pt.current_dims + pt.rank);
318 std::vector<size_t> chunk_dims(pt.chunk_dims, pt.chunk_dims + pt.rank);
319 std::vector<size_t> count(pt.count, pt.count + pt.rank);
320 std::vector<size_t> offset(pt.offset, pt.offset + pt.rank);
321
322 os << "rank: " << pt.rank << " N:" << pt.N <<" n:" << pt.n << "\n";
323 os << "element size: " << pt.element_size << " block size: " << pt.block_size << "\n";
324 os << "current dims: " << current_dims << std::endl;
325 os << "chunk dims: " << chunk_dims << std::endl;
326 os << "offset : " << offset << " count : " << count << std::endl;
327 os << "ds: "<< static_cast<hid_t>( pt.ds ) <<" dxpl: "<< static_cast<hid_t>( pt.dxpl ) << std::endl;
328 os << "fill value: " << std::hex << pt.fill_value << " buffer: " << pt.ptr;
329 os << "\n\n";
330 os << std::dec;
331 return os;
332}
