60 // actual implementation with full conversion allowed
61 template<class T, capi_close_t capi_close>
62 struct hid_t<T,capi_close, true,true,hdf5::any> {
63 using parent = hid_t<T,capi_close,true,true,hdf5::any>;
64 using hidtype = T;
65 // from CAPI
66 H5CPP__EXPLICIT hid_t( ::hid_t handle_ ) : handle( handle_ ){
67 if( H5Iis_valid( handle_ ) )
68 H5Iinc_ref( handle_ );
69 }
70 // TO CAPI
71 H5CPP__EXPLICIT operator ::hid_t() const {
72 return handle;
73 }
74 hid_t( std::initializer_list<::hid_t> fd ) // direct initialization doesn't increment handle
75 : handle( *fd.begin()){
76 }
77 hid_t() : handle(H5I_UNINIT){};
78 hid_t( const hid_t& ref) {
79 this->handle = ref.handle;
80 if( H5Iis_valid( handle ) )
81 H5Iinc_ref( handle );
82 }
83 hid_t& operator =( const hid_t& ref) {
84 if (this == &ref) return *this;
85 if( H5Iis_valid( handle ) )
86 capi_close( handle );
87 handle = ref.handle;
88 if( H5Iis_valid( handle ) )
89 H5Iinc_ref( handle );
90 return *this;
91 }
92 hid_t& operator =( hid_t&& ref) {
93 if (this == &ref) return *this;
94 if( H5Iis_valid( handle ) )
95 capi_close( handle );
96 handle = ref.handle;
97 ref.handle = H5I_UNINIT;
98 return *this;
99 }
100 /* move ctor must invalidate old handle */
101 hid_t( hid_t<T,capi_close,true,true,hdf5::any>&& ref ){
102 handle = ref.handle;
103 ref.handle = H5I_UNINIT;
104 }
105 ~hid_t(){
106 if( H5Iis_valid( handle ) )
107 capi_close( handle );
108 }
109 protected:
110 ::hid_t handle;
111 };
