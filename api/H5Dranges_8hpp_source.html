8 * Only available when compiled with -std=c++20 or later.

45 [[nodiscard]] inline hsize_t get_ds_element_count(h5::ds_t ds) {
46 hsize_t dims[H5CPP_MAX_RANK];
47 h5::sp_t fs = h5::get_space(ds);
48 int rank = static_cast<int>(h5::get_simple_extent_dims(fs, dims, nullptr));
49 if (rank != 1)
50 throw std::runtime_error("h5::view: only rank-1 datasets are supported");
51 return dims[0];
52 }

54 template<typename T>
55 struct iterator_t {
56 using value_type = T;
57 using difference_type = std::ptrdiff_t;
58 using iterator_category = std::input_iterator_tag;
59 using reference = const T&;
60 using pointer = const T*;
61
62 /* default-constructed iterator serves as the end sentinel */
63 iterator_t() : offset_(0), total_(0), chunk_n_(1) {}
64
65 /* Keyed constructor — only callable by code that holds a view_access key */
66 iterator_t(view_access, h5::ds_t ds, hsize_t offset, hsize_t total)
67 : offset_(offset), total_(total), chunk_n_(1)
68 {
69 if (offset_ >= total_)
70 return; // end sentinel — no pipeline needed
71
72 pipeline_ = std::make_shared<basic_pipeline_t>();
73 h5::dcpl_t dcpl = h5::get_dcpl(ds);
74 pipeline_->set_cache(dcpl, sizeof(T));
75 pipeline_->ds = ds;
76 pipeline_->dxpl = h5::default_dxpl;
77 chunk_n_ = pipeline_->n;
78 load_chunk();
79 }
80
81 reference operator*() const {
82 return reinterpret_cast<const T*>(pipeline_->chunk0)[offset_ % chunk_n_];
83 }
84 pointer operator->() const { return &**this; }
85
86 iterator_t& operator++() {
87 ++offset_;
88 if (offset_ < total_ && offset_ % chunk_n_ == 0)
89 load_chunk();
90 return *this;
91 }
92 iterator_t operator++(int) {
93 iterator_t tmp = *this;
94 ++(*this);
95 return tmp;
96 }
97
98 friend bool operator==(const iterator_t& a, const iterator_t& b) noexcept {
99 return a.offset_ == b.offset_;
100 }
101 friend bool operator!=(const iterator_t& a, const iterator_t& b) noexcept {
102 return !(a == b);
103 }
104
105 private:
106 void load_chunk() {
107 hsize_t chunk_offset = (offset_ / chunk_n_) * chunk_n_;
108 pipeline_->read_chunk(&chunk_offset, pipeline_->block_size, pipeline_->chunk0);
109 }
110
111 std::shared_ptr<basic_pipeline_t> pipeline_;
112 hsize_t offset_;
113 hsize_t total_;
114 hsize_t chunk_n_;
115 };

144 template<typename T>
145 [[nodiscard]] view_range<impl::iterator_t<T>> view(h5::ds_t ds) {
146 using Iter = impl::iterator_t<T>;
147 using Key = impl::view_access;
148 hsize_t n = impl::get_ds_element_count(ds);
149 return view_range<Iter>{ Iter(Key{}, ds, 0, n), Iter(Key{}, ds, n, n) };
150 }
