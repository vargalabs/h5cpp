143 template <class... args_t> inline
144 h5::current_dims_t get_current_dims(const h5::count_t& count, args_t&&... args ){
145 // premise: h5::current_dims{} is not present
146 using toffset = typename arg::tpos<const h5::offset_t&,const args_t&...>;
147 using tstride = typename arg::tpos<const h5::stride_t&,const args_t&...>;
148 using tblock = typename arg::tpos<const h5::block_t&,const args_t&...>;
149
150 h5::current_dims_t current_dims;
151 hsize_t rank = current_dims.rank = count.rank;
152
153 for(hsize_t i =0; i < rank; i++ )
154 current_dims[i] = count[i];
155 // TODO: following syntax looks better
156 //auto current_dims = static_cast<h5::current_dims_t>(count);
157
158 if constexpr( tstride::present ){ // when stride is not specified block == count
159 const h5::stride_t& stride = std::get<tstride::value>( std::forward_as_tuple( args...) );
160 if constexpr( tblock::present ){ // tricky, we have block as well
161 const h5::block_t& block = std::get<tblock::value>( std::forward_as_tuple( args...) );
162 for(hsize_t i=0; i < rank; i++)
163 current_dims[i] *= (stride[i] - block[i] + 1);
164 } else // block is not present, we are stretching `current_dims` with `stride`
165 for(hsize_t i=0; i < rank; i++)
166 current_dims[i] *= stride[i];
167 }
168 // we increment dimension with the specified offset, if any
169 if constexpr( toffset::present ){
170 const h5::offset_t& offset = std::get<toffset::value>( std::forward_as_tuple( args...) );
171 for(hsize_t i=0; i < rank; i++)
172 current_dims[i]+=offset[i];
173 }
174 return current_dims;
175 }
