13 (void)name; (void)size;
14 delete *static_cast< impl::pipeline_t<impl::basic_pipeline_t>**>( ptr );

17 // Copy callback for the high_throughput pipeline property.
18 //
19 // HDF5 ≥ 1.10.7 internally copies the DAPL when an object (dataset, file,
20 // attribute) is opened or created against it, so the object can carry its
21 // own DAPL. Without a copy callback, the property's bytes (the 8-byte
22 // pipeline pointer) are memcpy'd verbatim — leaving both the user's DAPL
23 // and HDF5's internal copy holding the same pipeline pointer. When each
24 // DAPL is later destroyed, dapl_pipeline_close fires on the same pointer
25 // twice → double-free. This is the root cause of the historic "crashes
26 // on 1.12.x" TODO. See issue #242.
27 //
28 // Fresh-allocation semantics are correct here because the pipeline is
29 // per-write scratch state, not shared accumulating state — set_cache()
30 // runs on every H5Dopen and on every h5::write/h5::read call.
31 inline herr_t dapl_pipeline_copy( const char *name, size_t size, void *value ){
32 (void)name; (void)size;
33 using pipeline = impl::pipeline_t<impl::basic_pipeline_t>;
34 *static_cast<pipeline**>(value) = new pipeline();
35 return 0;
36 }
37 inline ::herr_t dapl_pipeline_set(::hid_t dapl ) {
38 // ignore if already set
39 if( H5Pexist(dapl, H5CPP_DAPL_HIGH_THROUGHPUT) ) return 0;
40 using pipeline = impl::pipeline_t<impl::basic_pipeline_t>;
41 pipeline* ptr = new pipeline();
42 return H5Pinsert2(dapl, H5CPP_DAPL_HIGH_THROUGHPUT, sizeof( pipeline* ), &ptr,
43 nullptr, // set
44 nullptr, // get
45 nullptr, // prp_del
46 dapl_pipeline_copy, // copy — issue #242 fix
47 nullptr, // compare
48 dapl_pipeline_close);
49 }
50}
